import java.util.Random;

/**
 * @author haoli
 * @create 2020-04-12
 */
public class Producer {
    private Object lock;

    public Producer(Object lock) {
        this.lock = lock;
    }

    public void set(){
        try {
            synchronized (lock){
                //while value 不为null 等待
                while (Value.getValue() != null){
                    lock.wait();
                }
                //接受到通知后放入一个 通知所有线程
                Random random = new Random();
                Integer value = random.nextInt(20) + 1;
                Value.setValue(value);
                lock.notifyAll();
                System.out.printf("[%-5s][放入]:%-4d\n", Thread.currentThread().getName(), value);
            }
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}

消费者对象

/**
 * @author haoli
 * @create 2020-04-12
 */
public class Customer {
    private Object lock;

    public Customer(Object lock) {
        this.lock = lock;
    }

    public void get() {
        try {
            synchronized (lock) {
                //while Value为null，则等待
                while (Value.getValue() == null) {
                    lock.wait();
                }
                //接收到通知后 取出一个 通知所有线程
                Integer value = Value.getValue();
                Value.setValue(null);
                lock.notifyAll();
                System.out.printf("[%-5s][取出]:%-4d\n", Thread.currentThread().getName(), value);
            }
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}

/**
 * @author haoli
 * @create 2020-04-12
 */
public class ProducerRunnable implements Runnable {
    private Producer producer;

    public ProducerRunnable(Producer producer) {
        this.producer = producer;
    }


    @Override
    public void run() {
        try {
            while (true){
                producer.set();
                Thread.sleep(500);
            }
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}

public class App {

    public static void main(String[] args) {
        Object lock = new Object();
        Thread[] producerThreadArr = new Thread[5];
        Thread[] customerThreadArr = new Thread[5];

        for (int i = 0; i < 5; i++) {
            producerThreadArr[i] = new Thread(new ProducerRunnable(new Producer(lock)), "生产者" + (i + 1));
            customerThreadArr[i] = new Thread(new CustomerRunnable(new Customer(lock)), "消费者" + (i + 1));

            producerThreadArr[i].start();
            customerThreadArr[i].start();
        }
    }

}
